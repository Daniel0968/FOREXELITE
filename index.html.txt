      <h3>Soportes y Resistencias</h3>
      
      
      <p>Los soportes y resistencias son niveles de precio clave donde el mercado tiende a reaccionar de forma repetida. Un soporte es un nivel donde la presión compradora supera a la vendedora, provocando un rebote al alza. Una resistencia funciona de forma inversa.</p><p>Estos niveles se forman porque la memoria del mercado es colectiva: los participantes recuerdan dónde el precio reaccionó anteriormente y tienden a repetir ese comportamiento. Cuantas más veces se haya respetado un nivel, más significativo es.</p><p>Para identificarlos correctamente, busca zonas donde el precio haya tenido múltiples toques. Una zona es más fiable que una línea exacta. El timeframe en el que se forma también determina su importancia: un S/R en diario es más relevante que uno en 5 minutos.</p>

      <h3>Medias Móviles</h3>
      
      
      <p>Las medias móviles calculan el precio promedio durante un período determinado, suavizando el ruido del mercado y permitiendo identificar la dirección de la tendencia con mayor claridad.</p><p>Las más utilizadas son la EMA 20 (tendencia corto plazo), EMA 50 (tendencia media) y SMA 200 (tendencia principal). Cuando el precio cotiza por encima de la SMA 200, el mercado está en tendencia alcista de largo plazo.</p><p>El cruce de medias —cuando la EMA rápida cruza la EMA lenta— genera señales de posible cambio de tendencia. También actúan como soportes y resistencias dinámicas.</p>

      <h3>Order Blocks (OB)</h3>
      
      
      <p>Los Order Blocks son zonas de precio donde los participantes institucionales han ejecutado grandes órdenes, dejando un desequilibrio que el mercado eventualmente viene a corregir.</p><p>Un Bullish OB se forma como la última vela bajista antes de un impulso alcista significativo. Un Bearish OB es la última vela alcista antes de un impulso bajista fuerte. Cuando el precio regresa a estas zonas, los institucionales tienden a añadir más posiciones, generando una reacción predecible.</p><p>La clave está en combinar el OB con confluencias: zonas premium/discount, liquidez previa barrida y estructura de mercado favorable (BOS o CHoCH en timeframe superior).</p>

      <h3>Fair Value Gaps (FVG)</h3>
      
      
      <p>Un Fair Value Gap es un desequilibrio en el precio creado cuando el mercado se mueve tan rápido que deja un espacio entre velas consecutivas, una zona de ineficiencia sin transacciones reales.</p><p>El mercado tiende a rellenar estos gaps para equilibrar la oferta y la demanda. Los FVG son más fiables cuando aparecen después de una ruptura de estructura (BOS) y están alineados con la tendencia principal. Se pueden usar como zonas de entrada de alta precisión cuando coinciden con un OB.</p>

function rNews(){
  const el=document.getElementById('news-list');if(!el)return;
  const cu=S.cur?.u||'__g__';
  if(!S.news.length){el.innerHTML='<div class="nd-empty">Aún no hay noticias.<br>FX1000 publicará novedades aquí.</div>';updBadge('news');return;}
  el.innerHTML=[...S.news].reverse().map(n=>{
    const unr=!(n.readBy||[]).includes(cu);
    return`<div class="ni" onclick="expandN(${n.id},this)">
      <div class="ni-top"><span class="ni-ttl">${esc(n.title)}</span>
      <span class="ni-meta">${unr?'<span class="ni-unread"></span>':''}<span class="ni-date">${fmtD(n.date)}</span></span></div>
      <div class="ni-prev">${esc(n.body.slice(0,70))}${n.body.length>70?'…':''}</div>

        ${n.img?`<img src="${esc(n.img)}" alt="">`:''}
        ${n.body.split('\n\n').map(p=>`<p>${esc(p)}</p>`).join('')}
        ${isAdm()?`<button class="ni-del" onclick="event.stopPropagation();delNews(${n.id})">🗑 Eliminar</button>`:''}
      </div>

  }).join('');

}

function expandN(id,el){
  const b=document.getElementById('nib-'+id);if(!b)return;
  b.classList.toggle('on');
  if(b.classList.contains('on')&&S.cur){
    const n=S.news.find(x=>x.id===id);
    if(n&&!(n.readBy||[]).includes(S.cur.u)){if(!n.readBy)n.readBy=[];n.readBy.push(S.cur.u);sv();}
    el.querySelector('.ni-unread')?.remove();updBadge('news');
  }
}
function delNews(id){S.news=S.news.filter(x=>x.id!==id);sv();rNews();toast('🗑 Noticia eliminada');}

  const cu=S.cur?.u||'__g__';

}

function renderMd(txt){
  return txt.split('\n\n').map(p=>{
    if(p.startsWith('**')&&p.endsWith('**'))return`<h4 style="font-family:'Orbitron',monospace;font-size:.7rem;color:var(--gold);margin:.8rem 0 .35rem">${esc(p.slice(2,-2))}</h4>`;
    if(p.match(/^\d+\./))return`<p style="font-size:.78rem;color:var(--t2);line-height:1.75;margin-bottom:.55rem">${esc(p)}</p>`;
    return`<p style="font-size:.78rem;color:var(--t2);line-height:1.75;margin-bottom:.55rem">${esc(p)}</p>`;
  }).join('');
}

  document.getElementById('srch-ov').classList.remove('on');

}
function doSearch(){
  const q=document.getElementById('srch-in').value.trim().toLowerCase();
  const out=document.getElementById('srch-res');
  if(!q){out.innerHTML='';return;}
  const res=[];
  const chk=(items,sec,icon)=>items.forEach(a=>{
    const hay=(a.title||'')+(a.body||a.ana||'')+(a.pair||'')+(a.strat||'');
    if(hay.toLowerCase().includes(q)){
      const snip=hay.replace(/<[^>]+>/g,'').slice(0,90)+'…';
      res.push({sec:`${icon} ${sec}`,title:a.title||a.pair||'Entrada',snip,anchor:'s-'+sec.toLowerCase().split(' ')[0]});
    }
  });
  chk(S.tec,'Análisis Técnico','📈');
  chk(S.ins,'Análisis Institucional','🏦');
  chk(S.psy,'Psicología','🧠');
  chk(S.bt,'Backtesting','📊');
  S.news.forEach(n=>{if((n.title+n.body).toLowerCase().includes(q))res.push({sec:'📡 Noticias',title:n.title,snip:n.body.slice(0,80)+'…',anchor:null});});
  if(!res.length){out.innerHTML=`<div class="sr-zero">0 resultados para <strong style="color:var(--gold)">"${esc(q)}"</strong></div>`;return;}
  out.innerHTML=res.map(r=>`
    <div class="sr-card" onclick="goResult('${r.anchor}')">
      <div class="sr-sec">${esc(r.sec)}</div>
      <div class="sr-ttl">${esc(r.title)}</div>
      <div class="sr-snip">${esc(r.snip)}</div>
    </div>`).join('');
}
function goResult(anchor){
  closeSearch();
  if(anchor)setTimeout(()=>document.getElementById(anchor)?.scrollIntoView({behavior:'smooth'}),200);
}

const FOREX_PAIRS=[
  {label:'EUR/USD',base:'EUR',quote:'USD',decimals:4},
  {label:'GBP/USD',base:'GBP',quote:'USD',decimals:4},
  {label:'USD/JPY',base:'USD',quote:'JPY',decimals:2},

  {label:'CAD/JPY',base:'CAD',quote:'JPY',decimals:2},
  {label:'CHF/JPY',base:'CHF',quote:'JPY',decimals:2},
];

let tkRates={},tkPrices={},tkLoaded=false;

async function fetchForex(){
  try{
    // open.er-api.com: free, no key, CORS enabled, updates hourly
    const r=await fetch('https://open.er-api.com/v6/latest/USD');
    const d=await r.json();
    if(d.rates){
      tkRates=d.rates;
      computeForexPrices();
      tkLoaded=true;
      buildTicker(true);
      console.log('ForexElite: Precios Forex cargados desde API (actualización horaria)');
    }
  }catch(err){
    console.log('ForexElite: API no disponible, usando precios estimados');
    buildTicker(false);
  }
}

function computeForexPrices(){
  // rates from open.er-api: 1 USD = X currency
  const r=tkRates;
  const get=(base,quote)=>{
    // cross rate: how many quote per 1 base
    if(base==='USD')return r[quote]||1;
    if(quote==='USD')return r[base]?1/r[base]:1;
    return (r[base]&&r[quote])?(r[quote]/r[base]):null;
  };
  FOREX_PAIRS.forEach(p=>{
    const val=get(p.base,p.quote);
    if(val){
      tkPrices[p.label]={price:val,prev:val};
    }
  });
}

// Fallback prices (April 2026 estimates)
const FALLBACK={
  'EUR/USD':1.0432,'GBP/USD':1.2548,'USD/JPY':156.82,'USD/CHF':0.9118,
  'AUD/USD':0.6218,'USD/CAD':1.4392,'NZD/USD':0.5648,'EUR/GBP':0.8319,
  'EUR/JPY':163.71,'GBP/JPY':196.48,'EUR/CHF':0.9512,'AUD/JPY':97.55,
  'EUR/AUD':1.6775,'GBP/CHF':1.1432,'CAD/JPY':108.96,'CHF/JPY':171.99
};

function initTicker(){
  // Set fallback prices first so ticker shows immediately
  FOREX_PAIRS.forEach(p=>{tkPrices[p.label]={price:FALLBACK[p.label]||1,prev:FALLBACK[p.label]||1};});
  buildTicker(false);
  // Fetch real prices
  fetchForex();
  // Refresh every 60 seconds
  setInterval(fetchForex,60000);
  // Micro-movements every 2s to look live
  setInterval(simulateMicro,2000);
}

function simulateMicro(){
  FOREX_PAIRS.forEach(p=>{
    if(!tkPrices[p.label])return;
    const cur=tkPrices[p.label].price;
    const move=(Math.random()-.5)*cur*.0003;
    tkPrices[p.label].prev=cur;
    tkPrices[p.label].price=cur+move;
  });
  updateTickerDOM();
}

function buildTicker(live){
  const track=document.getElementById('tk');
  if(!track)return;
  // Build one full set, then duplicate for seamless loop
  const items=FOREX_PAIRS.map(p=>{
    const pr=tkPrices[p.label];
    const val=pr?pr.price.toFixed(p.decimals):FALLBACK[p.label]?.toFixed(p.decimals)||'–';
    const prev=pr?pr.prev:null;
    const up=prev?pr.price>=prev:true;
    const chg=prev&&prev>0?((pr.price-prev)/prev*100):0;
    const sign=chg>=0?'+':''
    return`<span class="tk-i" data-pair="${p.label}"><span class="p">${p.label}</span><span class="tv">${val}</span><span class="${up?'u':'d'} tc">${sign}${chg.toFixed(3)}%</span></span>`;
  }).join('');
  track.innerHTML=items+items; // duplicate for seamless loop
}

function updateTickerDOM(){
  document.querySelectorAll('.tk-i').forEach(el=>{
    const pair=el.dataset.pair;
    if(!pair||!tkPrices[pair])return;
    const p=FOREX_PAIRS.find(x=>x.label===pair);if(!p)return;
    const pr=tkPrices[pair];
    const val=pr.price.toFixed(p.decimals);
    const up=pr.price>=pr.prev;
    const chg=pr.prev>0?((pr.price-pr.prev)/pr.prev*100):0;
    const sign=chg>=0?'+':'';
    const tv=el.querySelector('.tv');const tc=el.querySelector('.tc');
    if(tv)tv.textContent=val;
    if(tc){tc.textContent=`${sign}${chg.toFixed(3)}%`;tc.className=`${up?'u':'d'} tc`;}
  });
}

  const candles=[];
  for(let i=0;i<100;i++){
    const o=base+(Math.random()-.5)*.0022,c=o+(Math.random()-.5)*.0028;
    const h=Math.max(o,c)+Math.random()*.0009,l=Math.min(o,c)-Math.random()*.0009;
    candles.push({o,c,h,l});base=c;
  }
  let off=0;
  function draw(){
    const W=cv.width,H=cv.height;
    ctx.clearRect(0,0,W,H);
    const prices=candles.flatMap(c=>[c.h,c.l]);
    const mn=Math.min(...prices),mx=Math.max(...prices),rng=mx-mn||.001;
    const toY=p=>H*.1+(H*.78)*(1-(p-mn)/rng);
    const cw=9,sp=15;
    candles.forEach((c,i)=>{
      const x=i*sp-off+W*.06;
      if(x<-11||x>W+11)return;
      const up=c.c>=c.o,col=up?'rgba(0,200,150,.6)':'rgba(255,69,96,.6)';
      ctx.strokeStyle=col;ctx.lineWidth=1;
      ctx.beginPath();ctx.moveTo(x,toY(c.h));ctx.lineTo(x,toY(c.l));ctx.stroke();
      ctx.fillStyle=col;
      const y1=toY(Math.max(c.o,c.c)),y2=toY(Math.min(c.o,c.c));
      ctx.fillRect(x-cw/2,y1,cw,Math.max(y2-y1,1));
    });
    const wn=14;ctx.strokeStyle='rgba(240,165,0,.28)';ctx.lineWidth=1.3;ctx.beginPath();
    candles.forEach((c,i)=>{
      if(i<wn)return;
      const avg=candles.slice(i-wn,i).reduce((s,x)=>s+x.c,0)/wn;
      const x=i*sp-off+W*.06;
      i===wn?ctx.moveTo(x,toY(avg)):ctx.lineTo(x,toY(avg));
    });
    ctx.stroke();
    off+=.25;
    if(off>sp){
      off-=sp;const last=candles[candles.length-1];
      const no=last.c,nc=no+(Math.random()-.5)*.003;
      candles.push({o:no,c:nc,h:Math.max(no,nc)+Math.random()*.001,l:Math.min(no,nc)-Math.random()*.001});
      candles.shift();
    }
    requestAnimationFrame(draw);
  }
  draw();
}

/* === SCROLL SPY === */
function initScrollSpy(){
  const secs=document.querySelectorAll('section:not(#hero)');
  const lks=document.querySelectorAll('#nb-links a[data-s]');
  const obs=new IntersectionObserver(entries=>{
    entries.forEach(e=>{if(e.isIntersecting){const id=e.target.id;lks.forEach(l=>l.classList.toggle('act',l.dataset.s===id));}});
  },{threshold:.2});
  secs.forEach(s=>obs.observe(s));
}

/* === FIU === */
function initFIU(){
  const obs=new IntersectionObserver(entries=>{
    entries.forEach(e=>{if(e.isIntersecting){e.target.classList.add('vis');obs.unobserve(e.target);}});
  },{threshold:.07});
  document.querySelectorAll('.fiu:not(.vis)').forEach(el=>obs.observe(el));
}
